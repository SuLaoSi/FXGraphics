CMAKE_MINIMUM_REQUIRED(VERSION 3.12.0 FATAL_ERROR)

IF(NOT DEFINED FXG_ROOT_DIR)
    MESSAGE(FATAL_ERROR "The project should be built from the top-level directory.")
ENDIF()

SET(PROJECT_NAME FXBasic)
PROJECT(${PROJECT_NAME})

SET(INCLUDE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/include)
SET(SRC_PATH ${CMAKE_CURRENT_SOURCE_DIR}/src)
SET(THIRDPARTY_PATH ${FXG_ROOT_DIR}/thirdparty)
SET(RESOURCE_PATH ${FXG_ROOT_DIR}/resources)

SET(INCLUDE_LIST
    ${INCLUDE_PATH}/basic_macro.h
    ${INCLUDE_PATH}/basic_platform.h
    ${INCLUDE_PATH}/basic_vector.h
    ${INCLUDE_PATH}/basic_log.h
)

SOURCE_GROUP(include FILES
    ${INCLUDE_LIST}
)

SET(SRC_LIST
    ${SRC_PATH}/basic_log.cpp
)

SOURCE_GROUP(src FILES
    ${SRC_LIST}
)

ADD_LIBRARY(${PROJECT_NAME}
    ${INCLUDE_LIST}
    ${SRC_LIST}
)

TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} PRIVATE
    ${INCLUDE_PATH}
    ${FXG_ROOT_DIR}/FXMath/include
)

TARGET_LINK_DIRECTORIES(${PROJECT_NAME} PRIVATE
    ${FXG_ROOT_DIR}/FXMath/include
)

TARGET_LINK_LIBRARIES(${PROJECT_NAME}
    FXMath
)
